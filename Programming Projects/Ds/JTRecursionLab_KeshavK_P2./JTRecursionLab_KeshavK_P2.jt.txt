// Calculates the power of a base
public class FancyCalc
{
	// Calculates the power of a base
	// @ param base the 'number to be powered
	// @ param exponent the power to which the base is calculated
	// @ return the calculated answer
	public pow(base, exponent)
	{
		if (exponent == 0) // base case
			return 1;
		else
         		 return this.pow(base, exponent-1)* base; // recursive case
	} 


	// Calculates the power of a base with lower computation time
	// @ param base the number to be powered
	// @ param exponent the power to which the base is calculated
	// @ return the calculated answer
	public fastPow(base, exponent)
	{
		
		if (exponent == 0) // base case
        			return 1;
		if (exponent % 2 == 0) // base case 2
			return this.square(
                		    this.fastPow(base, exponent/2));
     		return base * this.fastPow(base, exponent -1); // recursive case

	}
	// Calculates the square of a given number
	// @ param x the number being squared
	// @ return the square
	public square(x)
	{
		return x * x;
	}
	// Calculates the square of a given number
	// @ param x the number being squared
	// @ return the square
	public gcd(num1,num2)
	{
	 	
	 	if (num2 ==0) // base case
	    		return num1;
	 	return this.gcd(num2, num1 % num2); // recursive case
	}
	// Checks if a number is prime or not
	// @ param num the number being tested
	// @ return true if number is prime;
	// 	otherwise false
	public isPrime(num)
	{
		if (num <= 1) // base case
			return false;
		return this.helpPrime(num, 2); // recursive case
	}
	// Helps the isPrime method evaluate a number 
	// to check if it is prime by finding a number which
	// num is divisible by.
	// @ param num the number being tested
	// @ param divisor the first number that is checked
	// @ return a divisor is the number is not prime
	// 	otherwise false
	private helpPrime(num, divisor)
	{
		if (divisor == num) // base case
			return true;
		if (num % divisor == 0) // base case 2
			return false;
		return this.helpPrime(num,divisor+1); // recursive case
	}
	// Calculates the factorial of a number
	// @ param n the number which factorial is calculated for
	// @ return the calculated factorial
	public fact(n)
	{
		return this.factHelp(n- 1, n ); // recursive case
	}
	// Helper method for fact which assists in calculating 
	// the factorial
	// @ param n the number which factorial is calculated for
	// @ param result the value of the factorial calculations
	// @ return the calculated factorial
	private factHelp(n, result)
	{
		if (n == 0) // base case
		   return result; 
		if (n == -1) // base case
		   return 1;
		else
			return this.factHelp(n - 1, result * n ); // recursive case
	}
	// Finds the nth fibbonacci sequence number
	// @ param n the index of the number in the sequence
	// @ return the number
	public fib(n)
	{
	if (n == 0) // base case 1
		return 0;
	if (n == 1) // base case 2
		return 1;
	return this.fib(n-1) + this.fib(n-2); // recursive case
	}
	// Finds the square root of a number 
	// @ param n the number to be square rooted
	// @ return the square root
	public sqrt(n)
	{
	 return this.sqrthelp(n,n,n+1); // recursive case
	}
	// Finds the square root of a number by averaging the guess
	// with the number over the guess. Follows Newton's method
	// @ param n the index of the number in the sequence
	// @ param test the current guess 
	// @ param prevguess the previous guess
	// @ return the number
	private sqrthelp(n, test, prevguess)
	{
	 avg = (test+n/test)/2;
	 if (test == prevguess || test > prevguess) // base case
 	     return test;
	 return this.sqrthelp(n,avg,test); // recursive case
	}
}
